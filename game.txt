--- Guessing Game --- 

[START]

// Replay Loop //
WHILE True DO:
    // Initialize Game Variables
    SET targetNumber = Random Integer(1, 100)
    SET maxAttempts = 7
    SET attemptsLeft = maxAttempts
    SET hasWon = False

    PRINT "Welcome to the Number Guessing Game!"
    PRINT "There is a number between 1 and 100."
    PRINT "You have", [maxAttempts], "attempts to guess it correctly, good luck!."

// Core Game Loop //

WHILE attemptsLeft > 0 AND hasWon == False DO:
    PRINT "Attempts left:", attemptsLeft
    PROMPT user for guess
    READ userGuess

// Validate Input // 

IF userGuess is not a number DO:
    PRINT "Invalid input. Please enter a number."
    CONTINUE // Skip to next iteration of the loop

// Check Guesss // 

IF userGuess == targetNumber THEN:
    PRINT "Correct! You guessed the number!"
    SET hasWon = True
ELSE IF userGuess < targetNumber THEN:
    PRINT "Too low!"
ELSE:
    PRINT "Too high!"
        
// Decrement attempts // 
 SET attemptsLeft = attemptsLeft - 1

// Win/Loss Message // 
    IF hasWon == False THEN:
    PRINT "Game Over! The number was:", targetNumber

 // Replay Option // 
    PROMPT user "Would you like to play again? (yes/no)"
    READ playAgain
    IF playAgain != "yes" THEN:
        PRINT "Thanks for playing!"

BREAK // Break out of the main loop to exit

[END]
